53 // Evaluate Superquadric equation
54 double MySuperquadric::EvaluateFunction(double xyz[3])
55 {
56  double e = this->ThetaRoundness;
57  double n = this->PhiRoundness;
58  double p[3], s[3];
59  double val;
60 
61  s[0] = this->Scale[0] * this->Size;
62  s[1] = this->Scale[1] * this->Size;
63  s[2] = this->Scale[2] * this->Size;
64 
65  if (this->Toroidal) {
66  double tval;
67  double alpha;
68 
69  alpha = (1.0 / this->Thickness);
70  s[0] /= (alpha + 1.0);
71  s[1] /= (alpha + 1.0);
72  s[2] /= (alpha + 1.0);
73 
74  p[0] = (xyz[0] - this->Center[0]) / s[0];
75  p[1] = (xyz[1] - this->Center[1]) / s[1];
76  p[2] = (xyz[2] - this->Center[2]) / s[2];
77 
78  tval = pow((pow(fabs(p[2]), 2.0 / e) + pow(fabs(p[0]), 2.0 / e)), e / 2.0);
79  val = pow(fabs(tval - alpha), 2.0 / n) + pow(fabs(p[1]), 2.0 / n) - 1.0;
80  }
81  else { // Ellipsoidal
82  p[0] = (xyz[0] - this->Center[0]) / s[0];
83  p[1] = (xyz[1] - this->Center[1]) / s[1];
84  p[2] = (xyz[2] - this->Center[2]) / s[2];
85 
86  val = pow((pow(fabs(p[2]), 2.0 / e) + pow(fabs(p[0]), 2.0 / e)), e / n) +
87  pow(fabs(p[1]), 2.0 / n) - 1.0;
88  }
89 
90  if (val > MAX_FVAL){
91  val = MAX_FVAL;
92  }
93  else if (val < -MAX_FVAL){
94  val = -MAX_FVAL;
95  }
96 
97  return val;
98 }
